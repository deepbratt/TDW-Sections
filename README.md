## Install

```bash
npm install --save tdw-sections
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'tdw-sections'
import 'tdw-sections/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
